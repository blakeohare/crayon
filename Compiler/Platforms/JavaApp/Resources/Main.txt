package %%%JAVA_PACKAGE%%%;

import org.crayonlang.interpreter.Interpreter;
import org.crayonlang.interpreter.TranslationHelper;
import org.crayonlang.interpreter.ResourceReader;
import org.crayonlang.interpreter.structs.VmContext;

public final class Main {

  public static void main(String[] args) {
    TranslationHelper.setByteCode(ResourceReader.readFileText("resources/bytecode.txt"));
    TranslationHelper.setResourceManifest(ResourceReader.readFileText("resources/manifest.txt"));

    VmContext vm = Interpreter.v_createVm();
	Interpreter.v_vmEnvSetCommandLineArgs(vm, args);
    TranslationHelper.setProgramData(vm);
    Interpreter.v_startVm(vm);
  }

}
