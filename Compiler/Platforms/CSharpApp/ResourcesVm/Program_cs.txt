using System;
using System.Collections.Generic;
using System.Linq;
using Interpreter;
using Interpreter.Vm;

namespace CrayonRuntime
{
    internal class Program
    {
        private static readonly HashSet<string> COMPATIBLE_VERSIONS = new HashSet<string>() {
            "2.9.0",
        };

        public static void Main(string[] args)
        {
            try
            {
                Run(args);
            }
            catch (Exception e)
            {
                Console.Error.WriteLine(e.Message);
                if (e.InnerException != null)
                {
                    Console.Error.WriteLine("Cause:");
                    Console.Error.WriteLine(e.InnerException.Message);
                }
            }
        }

        private class RunCommand
        {
            // Errors
            public bool IncorrectNumArgs { get; set; }
            public bool InvalidParentProcessId { get; set; }
            public bool CbxPathNotValidFormat { get; set; }
            public bool CbxPathNonexistent { get; set; }

            public string CbxPath { get; set; }
            public bool UsePrefixes { get; set; }
            public bool ShowLibraryStack { get; set; }
            public int ParentProcessId { get; set; }
            public string[] RuntimeArgs { get; set; }
        }

        private static RunCommand GetRunCommand(string[] args)
        {
            RunCommand command = new RunCommand();
            if (args.Length < 2)
            {
                command.IncorrectNumArgs = true;
                return command;
            }

            List<string> argsList = new List<string>(args);
            command.CbxPath = argsList[0];

            Dictionary<string, string> vmArgs = new Dictionary<string, string>();
            for (int i = 1; i < args.Length; ++i)
            {
                string[] t = args[i].Split(':');
                string key = t[0];
                string value = t.Length > 1 ? t[1] : "";
                for (int j = 2; j < t.Length; ++j)
                {
                    value += ":" + t[j];
                }
                vmArgs[key] = value.Trim();
            }

            command.UsePrefixes = vmArgs.ContainsKey("showOutputPrefixes") && vmArgs["showOutputPrefixes"] == "yes";
            command.ShowLibraryStack = vmArgs.ContainsKey("showLibStack") && vmArgs["showLibStack"] == "yes";

            int p = 0;
            if (!vmArgs.ContainsKey("parentProcessId") || !int.TryParse(vmArgs["parentProcessId"], out p))
            {
                command.InvalidParentProcessId = true;
                return command;
            }
            command.ParentProcessId = p;

            command.RuntimeArgs = new string[0];
            if (vmArgs.ContainsKey("runtimeargs"))
            {
                string[] runtimeArgData = vmArgs["runtimeargs"].Split(':');
                if (runtimeArgData[0] != "0")
                {
                    command.RuntimeArgs = runtimeArgData[1]
                        .Split(',')
                        .Select(s => Base64Decode(s))
                        .ToArray();
                }
            }

            if (!command.CbxPath.ToLowerInvariant().EndsWith(".cbx")) command.CbxPathNotValidFormat = true;
            else if (!System.IO.File.Exists(command.CbxPath)) command.CbxPathNonexistent = true;

            return command;
        }

        private static void Run(string[] args)
        {
            RunCommand command = GetRunCommand(args);
            if (command.IncorrectNumArgs) throw new Exception("Incorrect number of args. Requires path to a cbx file and parent process ID.");
            if (command.InvalidParentProcessId) throw new Exception("Invalid or missing parent process ID.");
            if (command.CbxPathNotValidFormat) throw new Exception("File must be a CBX file.");
            if (command.CbxPathNonexistent) throw new Exception("Given CBX file does not exist.");

            if (command.ParentProcessId > 0)
            {
                // Start a thread that checks for the parent process. If the parent process disappears, then
                // the program should terminate.
                System.Threading.Thread thread = new System.Threading.Thread(ParentChecker);
                thread.IsBackground = true;
                thread.Start(command.ParentProcessId);
            }

            byte[] cbxData = System.IO.File.ReadAllBytes(command.CbxPath);

            CbxDecoder cbxDecoder = new CbxDecoder(cbxData);
            if (!COMPATIBLE_VERSIONS.Contains(cbxDecoder.CbxVersion))
                throw new Exception("CBX file was compiled with an incompatible version of Crayon than the current runtime.");

            string byteCode = cbxDecoder.ByteCode;
            if (byteCode == null) throw new Exception("CBX file did not contain byte code.");

            TranslationHelper.LibraryRegistry = new LibraryRegistry(cbxDecoder.LibraryVersionsByName);
            TranslationHelper.ImageSheetManifest = cbxDecoder.ImageSheetManifest;
            Interpreter.ResourceReader.CbxDirectory = System.IO.Path.GetDirectoryName(command.CbxPath);
            Interpreter.Structs.VmContext vm = CrayonWrapper.createVm(byteCode, cbxDecoder.ResourceManifest);
            CrayonWrapper.vmEnvSetCommandLineArgs(vm, command.RuntimeArgs);
            vm.environment.stdoutPrefix = command.UsePrefixes ? "STDOUT" : null;
            vm.environment.stacktracePrefix = command.UsePrefixes ? "STACKTRACE" : null;
            vm.environment.showLibStack = command.ShowLibraryStack;
            Debugger.INSTANCE.StartListening();
            TranslationHelper.ProgramData = vm;
            TranslationHelper.StartInterpreter();
        }

        private static string Base64Decode(string encodedValue)
        {
            byte[] utf8Bytes = System.Convert.FromBase64String(encodedValue);
            return System.Text.Encoding.UTF8.GetString(utf8Bytes);
        }

        // TODO: Surely there is a better way. But for now...
        private static void ParentChecker(object pid)
        {
            System.Diagnostics.Process parentProcess = System.Diagnostics.Process.GetProcessById((int)pid);
            while (true)
            {
                if (parentProcess == null || parentProcess.HasExited)
                {
                    System.Diagnostics.Process.GetCurrentProcess().Kill();
                    return; // not strictly necessary but prevents an infinite loop compiler warning.
                }
                System.Threading.Thread.Sleep(50);
            }
        }
    }
}
