# ExportStandaloneCbx(command, buildContext)

command = $1
buildContext = $2

outputDirectory = Exporter::GetOutputDirectory(buildContext)
compilationResult = Compile(buildContext)
resDb = GetResourceDatabaseFromBuild(buildContext)
byteCode = EncodeByteCode(compilationResult)
cbxFileBytes = GenerateCbxFileContent(buildContext, resDb, compilationResult, byteCode)
fileOutputContext = CreateFileOutputContext()
PopulateFileOutputContext(fileOutputContext, buildContext, resDb, cbxFileBytes)
EmitFilesToDisk(fileOutputContext, outputDirectory)

absoluteCbxFilePath = GetCbxFileLocation(outputDirectory, buildContext)

return absoluteCbxFilePath
